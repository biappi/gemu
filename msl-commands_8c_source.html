242 
243  return (((ge->rFO == JC_OPCODE) &&
244  ((M7 && !FA4 && !FA5) ||
245  (M6 && !FA4 && FA5) ||
246  (M5 && FA4 && !FA5) ||
247  (M4 && FA4 && FA5))) ||
