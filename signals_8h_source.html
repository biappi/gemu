207 static inline uint8_t NI_source(struct ge *ge, enum knot_ni_source source) {
208  uint8_t is_cn = ((ge->kNI.ni1 == NS_CN1) &&
209  (ge->kNI.ni2 == NS_CN2) &&
210  (ge->kNI.ni3 == NS_CN3) &&
211  (ge->kNI.ni4 == NS_CN4));
212 
213  uint16_t cn = is_cn ? ge_counting_network_output(ge) : 0;
214 
215  switch (source) {
216  case NS_CN1: return (cn & 0x000f) >> 0;
217  case NS_CN2: return (cn & 0x00f0) >> 4;
218  case NS_CN3: return (cn & 0x0f00) >> 8;
219  case NS_CN4: return (cn & 0xf000) >> 12;
220  case NS_RO1: return (ge->rRO & 0x0f) >> 0;
221  case NS_RO2: return (ge->rRO & 0xf0) >> 4;
222  case NS_UA2: return 0;
223  case NS_UA1: return 0;
224  }
